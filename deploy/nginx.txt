# nginx reverse proxy for SyncPad (HTTP only — TLS is added later with certbot).
#
# 1. The `map` below MUST live in the http { } context (e.g. in
#    /etc/nginx/nginx.conf or a file included from it), NOT inside a server { }
#    block. If your nginx already defines `$connection_upgrade`, skip this map.
# 2. Put the server { } block in /etc/nginx/sites-available/syncpad and symlink
#    it into /etc/nginx/sites-enabled/.
#
# SyncPad serves everything from one origin — the SPA, POST /api/docs, and the
# /ws/:id WebSocket — so a single proxied location covers it all. The upgrade
# headers make the WebSocket work; they are harmless for plain HTTP requests.

map $http_upgrade $connection_upgrade {
    default upgrade;
    ''      close;
}

server {
    listen 80;
    listen [::]:80;
    server_name syncpad.sholaayeni.xyz;

    # Modest cap; the server also enforces a 64 KB message limit itself.
    client_max_body_size 1m;

    location / {
        proxy_pass http://127.0.0.1:8090;
        proxy_http_version 1.1;

        # WebSocket upgrade.
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection $connection_upgrade;

        # Preserve the client's identity for the server's per-IP limit.
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Live sockets are long-lived; don't cut idle connections early.
        proxy_read_timeout 3600s;
        proxy_send_timeout 3600s;
    }
}

# Later, for TLS:
#   sudo certbot --nginx -d syncpad.sholaayeni.xyz
# certbot adds the listen 443 / ssl_certificate lines and an HTTP→HTTPS
# redirect. The client picks wss:// automatically once the page is HTTPS.
